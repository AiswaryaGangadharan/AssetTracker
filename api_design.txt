# ==============================
# EMPLOYEE FEATURES
# ==============================

Feature: Add a new employee
API Endpoint: POST /api/employees
Request Body:
  - name: str
  - employee_id: str
  - department: str
Response:
  - id: int
  - name: str
  - employee_id: str
  - department: str
Status Code: 201 Created

Feature: View all employees
API Endpoint: GET /api/employees
Response:
  - List of employees:
      - id: int
      - name: str
      - employee_id: str
      - department: str
Status Code: 200 OK

Feature: View single employee
API Endpoint: GET /api/employees/{id}
Response:
  - id: int
  - name: str
  - employee_id: str
  - department: str
Status Code: 200 OK

Feature: Update employee details
API Endpoint: PUT /api/employees/{id}
Request Body:
  - name: str
  - employee_id: str
  - department: str
Response:
  - id: int
  - name: str
  - employee_id: str
  - department: str
Status Code: 200 OK

Feature: Delete employee
API Endpoint: DELETE /api/employees/{id}
Response:
  - message: "Deleted successfully"
Status Code: 200 OK

# ==============================
# ASSET ASSIGNMENT FEATURES
# ==============================

Feature: Assign asset to employee
API Endpoint: POST /api/assignments
Request Body:
  - employee_id: int
  - asset_id: int
Response:
  - id: int
  - employee_id: int
  - asset_id: int
  - assigned_date: datetime
Status Code: 201 Created

Feature: Return asset
API Endpoint: PUT /api/assignments/{id}/return
Request Body:
  - return_date: datetime
Response:
  - id: int
  - employee_id: int
  - asset_id: int
  - assigned_date: datetime
  - return_date: datetime
Status Code: 200 OK

Feature: View assignment history
API Endpoint: GET /api/assignments
Response:
  - List of assignments:
      - id: int
      - employee_id: int
      - asset_id: int
      - assigned_date: datetime
      - return_date: datetime
Status Code: 200 OK
